```python
N = int(input())
memo = [0] * (N + 1)

def fibonacci(x):
    if x == 1 or x == 2:
        memo[x] = 1
    if memo[x] == 0:
        fibonacci(x - 1)
        memo[x] = memo[x-1] + memo[x-2]
    return (memo[x])

```

## Iterative Fibonacci

```python
N = int(input())

dp = [0] * (N + 1)
dp[1] = 1
dp[2] = 2

for i in range(3, N + 1);
	dp[i] = dp[i-1] + dp[i-2]

print(dp(N))
```
